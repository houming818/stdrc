
# Setup Neovim

## method 1 从release安装
wget https://github.com/neovim/neovim/releases/download/v0.10.0/nvim-linux64.tar.gz
tar xfvz nvim-linux64.tar.gz
mkdir ~/opt
mv nvim-linux64 ~/opt

### 编辑 PATH变量

PATH=$HOME/opt/nvim-linux64/bin:$PATH

### 必要安装以下插件
mkdir -p ~/.config/nvim/autoload
cp ~/.stdrc/std/init.vim ~/.config/nvim/init.vim
cp ~/.stdrc/std/plug.vim ~/.config/nvim/autoload/plug.vim
cp ~/.stdrc/std/coc-settings.json  ~/.config/nvim/
:PlugUpgrade
:PlugInstall

### 如果需要作为开发环境，需要安装以下插件
:CocInstall coc-css coc-git coc-go coc-html coc-java coc-json coc-lists coc-phpls coc-pyright coc-sh coc-tsserver coc-yaml coc-vetur
mkdir -p ~/.config/nvim/pack/github/start
git clone https://github.com/github/copilot.vim.git ~/.config/nvim/pack/github/start/copilot.vim
:Copilot setup
