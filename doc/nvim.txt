
# Setup Neovim

## method 1

apt install -y libncurses5-dev tree libxt-dev python-dev python3-dev ruby-dev lua5.1 liblua5.1-dev libperl-dev git build-essential clang python3-pip ycmd python3-hamcrest gperf luajit luarocks libuv1-dev libluajit-5.1-dev libunibilium-dev libmsgpack-dev libtermkey-dev libvterm-dev libutf8proc-dev gettext

pip install cmake

git clone https://github.com/neovim/neovim.git

cd neovim

git checkout v0.9.2

make CMAKE_BUILD_TYPE=Release CMAKE_INSTALL_PREFIX=/usr/local/bin

make install

cp nvim /usr/local/bin

## method 2

apt install -y lua5.1 liblua5.1-dev

curl https://cdn.grepcode.cn/cn.grepcode/neovim/nvim.amd64 > /tmp/nvim

chmod +x /tmp/nvim && mv /tmp/nvim /usr/local/bin/nvim

### 安装最新版的Neovim
sudo apt-get install software-properties-common
sudo add-apt-repository ppa:neovim-ppa/stable
sudo apt-get update
sudo apt-get install neovim

mkdir -p ~/.config/nvim/autoload
cp ~/.stdrc/std/init.vim ~/.config/nvim/init.vim
cp ~/.stdrc/std/plug.vim ~/.config/nvim/autoload/plug.vim
cp ~/.stdrc/std/coc-settings.json  ~/.config/nvim/
:CocInstall coc-css coc-git coc-go coc-html coc-java coc-json coc-lists coc-phpls coc-pyright coc-sh coc-tsserver coc-yaml coc-vue 
mkdir -p ~/.config/nvim/pack/github/start
git clone https://github.com/github/copilot.vim.git ~/.config/nvim/pack/github/start/copilot.vim
:Copilot setup
